
kubectl create namespace external-cluster

kubectl label namespace external-cluster istio-injection=enabled

cat <<EOT >> service-ext-deployment.yml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: service-ext-deployment
  labels:
    app: service-ext-app
    version: v1
spec:
  replicas: 1
  selector:
    matchLabels:
      app: service-ext-app
      version: v1
  template:
    metadata:
      labels:
        app: service-ext-app
        version: v1
    spec:
      containers:
        - name: service-ext-container
          image: artashesavetisyan/istio-basics-course:service-b
          imagePullPolicy: Always
          ports:
            - containerPort: 8082
          securityContext:
            runAsUser: 1000
EOT


kubectl apply -f service-ext-deployment.yml -n external-cluster

kubectl get pods --all-namespaces

cat <<EOT >> service-ext-srv.yml
apiVersion: v1
kind: Service
metadata:
  name: service-ext-srv
  labels:
    app: service-ext-app
    service: service-ext-app-srv
spec:
  ports:
    - port: 80
      name: http-80
      targetPort: 8082
  selector:
    app: service-ext-app
EOT

kubectl apply -f service-ext-srv.yml -n external-cluster

cat <<EOT >> service-ext-gw.yml
apiVersion: networking.istio.io/v1alpha3
kind: Gateway
metadata:
  name: service-ext-gw
spec:
  selector:
    istio: ingressgateway
  servers:
    - port:
        number: 80
        name: http
        protocol: HTTP
      hosts:
        - "*"
EOT

kubectl apply -f service-ext-gw.yml -n external-cluster

cat <<EOT >> inbound-to-service-ext-vs.yml
apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: inbound-to-service-ext-vs
spec:
  hosts:
    - "*"
  gateways:
    - service-ext-gw
  http:
    - match:
        - uri:
            exact: /service-ext
      rewrite:
        uri: /
      route:
        - destination:
            host: service-ext-srv
            port:
              number: 80
EOT

kubectl apply -f inbound-to-service-ext-vs.yml -n external-cluster

curl -v http://$GATEWAY_URL/service-ext

kubectl describe service istio-ingressgateway -n istio-system

curl -v http://istio-ingressgateway.istio-system.svc.cluster.local:80/service-ext
